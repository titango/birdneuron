Usage

Create Bird model

class Bird {
	      constructor()
	      {
			this.x = 64;
			this.y = 10;
	      }
	
	      copy() // Required!
	      {
			return new Bird();
	      }
		
		// function that returns an array of input numbers
		think()
		{
			. . . . .
		}
	
		// function that will determine what the bird will do 
// when receiving an output array
		do(outputs)
		{
			. . . . . .
		}

		// Display or draw the bird
		show()
		{
			. . . . . .
		}
}
*Note that function copy() is required which return new model object.

Usage

Create BirdNeuron object and insert models in main file


//Using bird neuron
let bn = new BirdNeuron();

//Settings
bn.totalPopulation = 500;
bn.inputlayer = 5;
bn.hiddenlayer = 8;
bn.outputlayer = 2;

//Create bird models
      var tempBirds = [];
      for (let i = 0; i < bn.totalPopulation; i++) {
        let bird = new Bird();
        // console.log("bird: ", bird);
        tempBirds.push(bird);
      }
	
	//Insert to both population and total population
      bn.input(tempBirds, false);

	//Models in action - show the birds
for (let i = bn.activePopulation.length - 1; i >= 0; i--) {
          let bird = bn.activePopulation[i];

          // Bird uses its brain!
          bird.inputs = bird.think();

          var actions = bird.outputs();

          if(actions) bird.do(actions);
          bird.show();
